# i18n Refactor Phase 1 — Implementation Plan

**Goal:** Replace `packages/i18n` internals with react-i18next while preserving the identical public API — zero changes to any consumer file.

**Architecture:** Wrapper pattern. `useTranslation()` and `TranslationProvider` keep the same signatures but delegate to react-i18next internally. Translation files convert from TypeScript objects to feature-based JSON namespaces. Custom scripts provide type generation and sync checking.

**Tech Stack:** i18next, react-i18next, i18next-icu, i18next-resources-to-backend, tsx (scripts)

**Spec:** `docs/i18n-refactor/design-spec.md`
**ADR:** `docs/i18n-refactor/research-adr.md`

### Known Behavior Change

**Language switching flash:** The current implementation pre-loads all 19 languages in the background, so switching is instant. After the refactor, only the current language + fallback are loaded. Switching to a new language will briefly show raw key strings (~100-300ms) while the new language's JSON files are fetched, then re-render. This is an accepted trade-off — it eliminates ~5 MB of background loading in exchange for a brief flash on language switch (an infrequent action).

async function convertLanguage(lang: string): Promise<void> {
  const langDir = path.join(LOCALES_DIR, lang);

  // Import translations.ts
  const translationsMod = await import(path.join(langDir, "translations.ts"));
  const translations = translationsMod.default;

  // Split translations by namespace
  const namespaces: Record<string, Record<string, unknown>> = {};

  for (const [key, value] of Object.entries(translations)) {
    const ns = typeof value === "string" ? "common" : (NAMESPACE_MAP[key] || "common");
    if (!namespaces[ns]) namespaces[ns] = {};
    namespaces[ns][key] = value;
  }

  // Import and convert standalone files
  const standaloneFiles = ["editor", "empty-state", "accessibility", "tour"];
  for (const file of standaloneFiles) {
    try {
      const tsFile = file === "empty-state" ? "empty-state" : file;
      const mod = await import(path.join(langDir, `${tsFile}.ts`));
      namespaces[file] = mod.default;
    } catch {
      console.warn(`  Warning: ${file}.ts not found for ${lang}`);
    }
  }

  // Write JSON files
  for (const [ns, data] of Object.entries(namespaces)) {
    const jsonPath = path.join(langDir, `${ns}.json`);
    fs.writeFileSync(jsonPath, JSON.stringify(data, null, 2) + "\n");
  }

  // Count keys
  const totalKeys = Object.values(namespaces).reduce(
    (sum, ns) => sum + countKeys(ns),
    0
  );
  console.log(`  ${lang}: ${totalKeys} keys across ${Object.keys(namespaces).length} namespaces`);
}

function countKeys(obj: Record<string, unknown>, prefix = ""): number {
  let count = 0;
  for (const [key, value] of Object.entries(obj)) {
    if (typeof value === "object" && value !== null) {
      count += countKeys(value as Record<string, unknown>, `${prefix}${key}.`);
    } else {
      count++;
    }
  }
  return count;
}

async function main() {
  console.log("Converting TS translations to JSON namespaces...\n");

  for (const lang of SUPPORTED_LANGUAGES) {
    await convertLanguage(lang);
  }

  console.log("\nDone. Now delete the old .ts files and run sync-check.");
}

main().catch(console.error);

// packages/i18n/src/provider/index.tsx

import React, { useEffect, useState } from "react";
import { I18nextProvider } from "react-i18next";
import { i18nInstance, initPromise } from "../core";
import { FALLBACK_LANGUAGE, LANGUAGE_STORAGE_KEY } from "../constants/language";

interface TranslationProviderProps {
  children: React.ReactNode;
}

export const TranslationProvider: React.FC<TranslationProviderProps> = ({ children }) => {
  const [isReady, setIsReady] = useState(i18nInstance.isInitialized);

  useEffect(() => {
    initPromise
      .then(() => {
        if (typeof window === "undefined") return;
        const savedLocale = localStorage.getItem(LANGUAGE_STORAGE_KEY) || FALLBACK_LANGUAGE;
        return i18nInstance.changeLanguage(savedLocale);
      })
      .then(() => setIsReady(true))
      .catch((err) => {
        console.error("Failed to initialize i18n:", err);
        setIsReady(true);
      });
  }, []);

  if (!isReady) return null;

  return <I18nextProvider i18n={i18nInstance}>{children}</I18nextProvider>;
};

// packages/i18n/src/hooks/use-translation.ts

import { useTranslation as useI18nextTranslation } from "react-i18next";
import { SUPPORTED_LANGUAGES, LANGUAGE_STORAGE_KEY } from "../constants/language";
import { NAMESPACES } from "../constants/namespaces";
import type { TLanguage, ILanguageOption } from "../types";

export type TTranslationStore = {
  t: (key: string, params?: Record<string, unknown>) => string;
  currentLocale: TLanguage;
  changeLanguage: (lng: TLanguage) => void;
  languages: ILanguageOption[];
};

export function useTranslation(): TTranslationStore {
  const { t, i18n } = useI18nextTranslation(NAMESPACES as unknown as string[]);

  return {
    t: (key: string, params?: Record<string, unknown>) => t(key, params as Record<string, string>),
    currentLocale: i18n.language as TLanguage,
    changeLanguage: (lng: TLanguage) => {
      i18n.changeLanguage(lng).then(() => {
        if (typeof window === "undefined") return;
        localStorage.setItem(LANGUAGE_STORAGE_KEY, lng);
        document.documentElement.lang = lng;
      });
    },
    languages: SUPPORTED_LANGUAGES,

```typescript
// packages/i18n/scripts/generate-types.ts
//
// Reads all English JSON namespace files, flattens keys with dot notation,
// and writes src/types/keys.generated.ts with a union type of all keys.
//
// Usage: npx tsx packages/i18n/scripts/generate-types.ts

import * as fs from "node:fs";
import * as path from "node:path";

const LOCALES_DIR = path.resolve(import.meta.dirname, "../src/locales");
const OUTPUT_FILE = path.resolve(import.meta.dirname, "../src/types/keys.generated.ts");
const SOURCE_LANG = "en";

function flattenKeys(obj: Record<string, unknown>, prefix = ""): string[] {
  const keys: string[] = [];
  for (const [key, value] of Object.entries(obj)) {
    const fullKey = prefix ? `${prefix}.${key}` : key;
    if (typeof value === "object" && value !== null && !Array.isArray(value)) {
      keys.push(...flattenKeys(value as Record<string, unknown>, fullKey));
    } else {
      keys.push(fullKey);
    }
  }
  return keys;
}

function main() {
  const langDir = path.join(LOCALES_DIR, SOURCE_LANG);
  const jsonFiles = fs.readdirSync(langDir).filter((f) => f.endsWith(".json"));

  const allKeys = new Set<string>();
  const keysByNamespace = new Map<string, string[]>();

  for (const file of jsonFiles) {
    const filePath = path.join(langDir, file);
    const content = JSON.parse(fs.readFileSync(filePath, "utf-8"));
    const keys = flattenKeys(content);
    const namespace = file.replace(".json", "");
    keysByNamespace.set(namespace, keys);
    keys.forEach((k) => allKeys.add(k));
  }

  // Check for cross-namespace collisions
  const keyToNamespaces = new Map<string, string[]>();
  for (const [ns, keys] of keysByNamespace) {
    for (const key of keys) {
      const existing = keyToNamespaces.get(key) || [];
      existing.push(ns);
      keyToNamespaces.set(key, existing);
    }
  }

  const collisions = [...keyToNamespaces.entries()].filter(([, nss]) => nss.length > 1);
  if (collisions.length > 0) {
    console.error("\nCROSS-NAMESPACE COLLISIONS DETECTED:");
    for (const [key, nss] of collisions) {
      console.error(`  "${key}" exists in: ${nss.join(", ")}`);
    }
    console.error("\nFix collisions before generating types.");
    process.exit(1);
  }

  // Check for path conflicts (key is both a leaf and a branch)
  const sortedKeys = [...allKeys].sort();
  for (let i = 0; i < sortedKeys.length - 1; i++) {
    if (sortedKeys[i + 1].startsWith(sortedKeys[i] + ".")) {
      console.error(`\nPATH CONFLICT: "${sortedKeys[i]}" is a leaf but "${sortedKeys[i + 1]}" extends it as a branch.`);
      process.exit(1);
    }
  }

  // Generate the type file
  const header = [
    "/**",
    " * SPDX-FileCopyrightText: 2023-present Plane Software, Inc.",
    " * SPDX-License-Identifier: LicenseRef-Plane-Commercial",
    " *",
    ' * Licensed under the Plane Commercial License (the "License");',
    " * you may not use this file except in compliance with the License.",
    " * You may obtain a copy of the License at",
    " * https://plane.so/legals/eula",
    " *",
    " * DO NOT remove or modify this notice.",
    " * NOTICE: Proprietary and confidential. Unauthorized use or distribution is prohibited.",
    " */",
    "",
    "// AUTO-GENERATED — DO NOT EDIT",
    `// Generated from ${jsonFiles.length} English namespace files (${allKeys.size} keys)`,
    "// Run: pnpm run generate:types",
    "",
  ].join("\n");

  const typeBody = sortedKeys.map((key) => `  | "${key}"`).join("\n");
  const content = `${header}export type TTranslationKeys =\n${typeBody};\n`;

  fs.writeFileSync(OUTPUT_FILE, content);

  console.log(`Generated ${allKeys.size} keys from ${jsonFiles.length} namespace files.`);
  console.log(`Output: ${OUTPUT_FILE}`);
}

main();
```

```typescript
// packages/i18n/scripts/sync-check.ts
//
// Compares translation keys across all locales against English (source).
// Detects: missing keys, stale keys, cross-namespace collisions, path conflicts.
//
// Usage:
//   npx tsx packages/i18n/scripts/sync-check.ts          # Report only
//   npx tsx packages/i18n/scripts/sync-check.ts --ci      # Exit 1 if issues found

import * as fs from "node:fs";
import * as path from "node:path";

const LOCALES_DIR = path.resolve(import.meta.dirname, "../src/locales");
const SOURCE_LANG = "en";
const CI_MODE = process.argv.includes("--ci");

const SUPPORTED_LANGUAGES = [
  "en", "fr", "es", "ja", "zh-CN", "zh-TW", "ru", "it", "cs", "sk",
  "de", "ua", "pl", "ko", "pt-BR", "id", "ro", "vi-VN", "tr-TR",
];

function flattenKeys(obj: Record<string, unknown>, prefix = ""): string[] {
  const keys: string[] = [];
  for (const [key, value] of Object.entries(obj)) {
    const fullKey = prefix ? `${prefix}.${key}` : key;
    if (typeof value === "object" && value !== null && !Array.isArray(value)) {
      keys.push(...flattenKeys(value as Record<string, unknown>, fullKey));
    } else {
      keys.push(fullKey);
    }
  }
  return keys;
}

function getLocaleKeys(lang: string): { keys: Set<string>; keysByFile: Map<string, string[]> } {
  const langDir = path.join(LOCALES_DIR, lang);
  const jsonFiles = fs.readdirSync(langDir).filter((f) => f.endsWith(".json"));
  const keys = new Set<string>();
  const keysByFile = new Map<string, string[]>();

  for (const file of jsonFiles) {
    const content = JSON.parse(fs.readFileSync(path.join(langDir, file), "utf-8"));
    const fileKeys = flattenKeys(content);
    keysByFile.set(file, fileKeys);
    fileKeys.forEach((k) => keys.add(k));
  }

  return { keys, keysByFile };
}

function main() {
  let hasIssues = false;

  // Get English keys as source of truth
  const en = getLocaleKeys(SOURCE_LANG);
  const enKeys = en.keys;

  console.log("=== Sync Check Results ===\n");
  console.log(`  ${SOURCE_LANG}:    ${enKeys.size} keys (source)\n`);

  // Cross-namespace collision check (English only)
  const keyToFiles = new Map<string, string[]>();
  for (const [file, keys] of en.keysByFile) {
    for (const key of keys) {
      const existing = keyToFiles.get(key) || [];
      existing.push(file);
      keyToFiles.set(key, existing);
    }
  }

  const collisions = [...keyToFiles.entries()].filter(([, files]) => files.length > 1);
  if (collisions.length > 0) {
    hasIssues = true;
    console.log("CROSS-NAMESPACE COLLISIONS:");
    for (const [key, files] of collisions) {
      console.log(`  ✗ "${key}" exists in: ${files.join(", ")}`);
    }
    console.log();
  }

  // Path conflict check (English only)
  const sortedKeys = [...enKeys].sort();
  const pathConflicts: string[] = [];
  for (let i = 0; i < sortedKeys.length - 1; i++) {
    if (sortedKeys[i + 1].startsWith(sortedKeys[i] + ".")) {
      pathConflicts.push(`"${sortedKeys[i]}" is a leaf but "${sortedKeys[i + 1]}" extends it`);
    }
  }

  if (pathConflicts.length > 0) {
    hasIssues = true;
    console.log("PATH CONFLICTS:");
    for (const conflict of pathConflicts) {
      console.log(`  ✗ ${conflict}`);
    }
    console.log();
  }

  // Per-locale comparison
  const missingByLocale = new Map<string, string[]>();

  for (const lang of SUPPORTED_LANGUAGES) {
    if (lang === SOURCE_LANG) continue;

    const locale = getLocaleKeys(lang);
    const missing = [...enKeys].filter((k) => !locale.keys.has(k));
    const stale = [...locale.keys].filter((k) => !enKeys.has(k));
    const pct = ((locale.keys.size / enKeys.size) * 100).toFixed(1);

    if (missing.length > 0) {
      missingByLocale.set(lang, missing);
      hasIssues = true;
    }

    const status = missing.length === 0 && stale.length === 0 ? "✓" : "✗";
    console.log(
      `  ${status} ${lang.padEnd(8)} ${locale.keys.size} keys (${pct}%)` +
        (missing.length > 0 ? ` — ${missing.length} missing` : "") +
        (stale.length > 0 ? ` — ${stale.length} stale` : "")
    );
  }

  // Print missing key details
  if (missingByLocale.size > 0) {
    console.log("\n--- Missing Keys Detail ---\n");
    for (const [lang, missing] of missingByLocale) {
      console.log(`${lang} (${missing.length} missing):`);
      for (const key of missing.slice(0, 20)) {
        console.log(`  - ${key}`);
      }
      if (missing.length > 20) {
        console.log(`  ... and ${missing.length - 20} more`);
      }
      console.log();
    }
  }

  // Exit code
  if (CI_MODE && hasIssues) {
    console.log("\nCI MODE: Issues found. Exiting with code 1.");
    process.exit(1);
  }
}

main();
```
